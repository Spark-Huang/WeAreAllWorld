### 3.2 Kubernetes资源配置

#### Bot Deployment

```yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: telegram-bot
  namespace: we-are-all-world
spec:
  replicas: 2
  selector:
    matchLabels:
      app: telegram-bot
  template:
    metadata:
      labels:
        app: telegram-bot
    spec:
      containers:
        - name: bot
          image: your-registry/telegram-bot:latest
          env:
            - name: TELEGRAM_BOT_TOKEN
              valueFrom:
                secretKeyRef:
                  name: bot-secrets
                  key: telegram-token
            - name: SUPABASE_URL
              valueFrom:
                secretKeyRef:
                  name: supabase-secrets
                  key: url
            - name: SUPABASE_SERVICE_KEY
              valueFrom:
                secretKeyRef:
                  name: supabase-secrets
                  key: service-key
          resources:
            limits:
              memory: "512Mi"
              cpu: "500m"
            requests:
              memory: "256Mi"
              cpu: "250m"
```

#### OpenClaw Pod模板

```yaml
# 由管理沙箱动态创建
apiVersion: apps/v1
kind: Deployment
metadata:
  name: openclaw-${USER_ID}
  namespace: we-are-all-world
spec:
  replicas: 1
  template:
    spec:
      containers:
        - name: openclaw
          image: ghcr.io/openclaw/openclaw:latest
          env:
            - name: OPENCLAW_WORKSPACE_DIR
              value: /home/node/.openclaw/workspace
            - name: LLM_API_KEY
              valueFrom:
                secretKeyRef:
                  name: llm-secrets
                  key: api-key
          volumeMounts:
            - mountPath: /home/node/.openclaw
              name: workspace
      volumes:
        - name: workspace
          persistentVolumeClaim:
            claimName: openclaw-pvc-${USER_ID}
```

export function logInfo(action: string, message: string, metadata?: any) {
  console.log(JSON.stringify({
    timestamp: new Date().toISOString(),
    level: 'info',
    action,
    message,
    metadata
  }));

  console.error(JSON.stringify({

## 7. 风险与应对

### 快速部署脚本

```bash
#!/bin/bash
# deploy.sh

echo "部署 共生世界（WeAreAll.World） MVP..."

# 1. 创建命名空间
kubectl create namespace we-are-all-world

# 2. 创建Secrets
kubectl create secret generic bot-secrets \
  --from-literal=telegram-token=$TELEGRAM_BOT_TOKEN \
  -n we-are-all-world

kubectl create secret generic supabase-secrets \
  --from-literal=url=$SUPABASE_URL \
  --from-literal=service-key=$SUPABASE_SERVICE_KEY \
  -n we-are-all-world

kubectl create secret generic llm-secrets \
  --from-literal=api-key=$OPENAI_API_KEY \
  -n we-are-all-world

# 3. 部署Bot
kubectl apply -f k8s/telegram-bot-deployment.yaml

# 4. 部署定时任务
kubectl apply -f k8s/cronjob-weekly-evaluation.yaml
kubectl apply -f k8s/cronjob-dormant-decay.yaml

echo "部署完成！"
```
